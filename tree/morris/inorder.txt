private void morris(TreeNode root) {		
	TreeNode p = root, pre;
	while (p != null) {
		if (p.left == null) {
			System.out.print(p.val);
			p = p.right;
		} else {
			pre = p.left;
			while (pre.right != null && pre.right != p) pre = pre.right;
			if (pre.right == null) {
				pre.right = p;
				p = p.left;
			} else {
				pre.right = null;
				System.out.print(p.val);
				p = p.right;
			}
		}
	}
}
